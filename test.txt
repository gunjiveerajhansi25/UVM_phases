class my_test extends uvm_test;
  
  `uvm_component_utils(my_test)
  
  function new(string name ="m_test", uvm_component parent = null);
    super.new(name,parent);
  endfunction
  
  my_env m_env;
  my_sequence m_seq;
  
  virtual function void build_phase(uvm_phase phase);
    super.build_phase(phase);
    `uvm_info("TEST","Starting of build_phase...",UVM_MEDIUM)
    m_env = my_env ::type_id::create("m_env",this);
    m_seq = my_sequence :: type_id :: create("m_seq");
  endfunction
  
  virtual function void connect_phase(uvm_phase phase);
   super.connect_phase(phase);
    `uvm_info("TEST","Starting of connect_phase...",UVM_MEDIUM)
  endfunction
  
  virtual function void end_of_elaboration_phase(uvm_phase phase);
    super.end_of_elaboration_phase(phase);
    `uvm_info("TEST","Starting of end_of_elaboration_phase..",UVM_MEDIUM)
  endfunction
  
  virtual function void start_of_simulation_phase(uvm_phase phase);
    super.start_of_simulation_phase(phase);
    `uvm_info("TEST","Starting of start_of_simulation_phase..",UVM_MEDIUM)
    // uvm_top.print_topology();
  endfunction
  
  virtual task run_phase(uvm_phase phase);
    super.run_phase(phase);
    `uvm_info("TEST","Starting of run_phase...",UVM_MEDIUM)
    phase.raise_objection(this);
    m_seq.start(m_env.m_age.m_seqr);
    phase.drop_objection(this);
    `uvm_info("TEST","END of run_phase...",UVM_MEDIUM)
  endtask
  
  virtual function void extract_phase(uvm_phase phase);
    super.extract_phase(phase);
    `uvm_info("TEST","Starting of extract..",UVM_MEDIUM)
  endfunction
  
    virtual function void check_phase(uvm_phase phase);
   super.check_phase(phase);
      `uvm_info("TEST","Starting of check_phase...",UVM_MEDIUM)
  endfunction
  
    virtual function void report_phase(uvm_phase phase);
    super.report_phase(phase);
      `uvm_info("TEST","Starting of report..",UVM_MEDIUM)
  endfunction
  
    virtual function void final_phase(uvm_phase phase);
   super.final_phase(phase);
      `uvm_info("TEST","Starting of final_phase...",UVM_MEDIUM)
  endfunction
  
  
  
endclass
